ALTER PROCEDURE [dbo].[Card_History_CopyFromCard]
	@New_Card_ID nvarchar(20),
	@Old_Card_ID nvarchar(20),
	@Type int
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;
    INSERT INTO Card_History
				select @New_Card_ID, Card_ID, @Type,Lead_ID,[Programme_ID]
					  ,[Is_Primary]
					  ,[Name_On_Card]
					  ,[CardType_ID]
					  ,[Sold_Date]
					  ,[Expiry_Date]
					  ,[Member_Since]
					  ,[Register_Date]
					  ,[Cancel_Date]
					  ,[Upgraded_Date]
					  ,[Renewal]
					  ,[Renewal_Date]
					  ,[Printing_Date]
					  ,[Issued_Date]
					  ,[Refund_Date]
					  ,[Last_Status_Changed]
					  ,[SalesPerson]
					  ,[Delivery_By]
					  ,[Introduced_By]
					  ,[Payment_ID]
					  ,[CreditCard]
					  ,[Reference]
					  ,[Expiry_CreditCard]
					  ,[Week]
					  ,[Price]
					  ,[Currency]
					  ,[Currency_Rate]
					  ,[Status_ID]
					  ,[Total_Points]
					  ,[Total_Nights]
					  ,[Total_Revenue]
					  ,[PointBalance]
					  ,[RoomRevenue]
					  ,[SeminaRevenue]
					  ,[BookingRevenue]
					  ,[FoodBeverage]
					  ,[OtherRevenue]
					  ,[Approver_Unused]
					  ,[Approver_Issued]
					  ,[Approver_Expired]
					  ,[Approver_Printing]
					  ,[Approver_Awaiting]
					  ,[Approver_Cancel]
					  ,[Approver_Renew]
					  ,[Approver_Refund]
					  ,[Created_By]
					  ,[Created_Date]
					  ,[Changed_By]
					  ,[Changed_Date]
					  ,[Active]
					  ,null
					  ,[Benefit_Renewal]
					  ,[Note]
					  from MemberCards where Card_ID=@Old_Card_ID
END
