ALTER FUNCTION [dbo].[func_GetGuestCallBySpouseDOB]

(

@before DATE, @after DATE, @isSpouseDOBChecked BIT, @isActive BIT, @isInactive

BIT, @CardTypeList VARCHAR(100)

)

RETURNS @tempTable TABLE

(

Lead_ID INT, Card_ID NVARCHAR(20), Sold_Date DATE, Lead_Name VARCHAR(60),

Mobile_No VARCHAR(30), Birthday DATE, Anniversary DATE, SpouseDOB

DATE, IsEU BIT, IsBD BIT, IsSpouse BIT, IsWedding BIT, Is4Month BIT,

Is8Month BIT, Is6Month BIT, IsExpDate BIT, Status_ID INT

)

AS

BEGIN

IF @isSpouseDOBChecked = 1

BEGIN

DECLARE @setting VARCHAR(8) = (

SELECT SettingValue

FROM ADM_Settings

WHERE Setting_ID =

'flgMemberCards_GuestServiceCallCardStatus'

)

--select @setting

DECLARE @Temp TABLE(Card_ID VARCHAR(20))

DECLARE @isIssued BIT = CONVERT(BIT, SUBSTRING(@setting, 4, 1)) --//Issued

DECLARE @isCancel BIT = CONVERT(BIT, SUBSTRING(@setting, 5, 1))--//Canceled

DECLARE @isExp BIT = CONVERT(BIT, SUBSTRING(@setting, 6, 1))--//Expiry

DECLARE @isRef BIT = CONVERT(BIT, SUBSTRING(@setting, 7, 1))--//Refund

DECLARE @isReprint BIT = CONVERT(BIT, SUBSTRING(@setting, 8, 1))--//Reprint

DECLARE @cardTypeTblTemp AS TABLE (CardType_ID VARCHAR(10))

INSERT INTO @cardTypeTblTemp

SELECT *

FROM [dbo].[func_InsertStringToTable](@CardTypeList)

IF dbo.func_CheckSettingBoolean('flgMemberCards_GuestServiceCallExcludeHistory')= 0

BEGIN

INSERT INTO @tempTable

SELECT Lead_ID, Card_ID, Sold_Date, Lead_Name, Mobile_No, Birthday,

Anniversary, SpouseDOB, 0, 0, 1, 0, 0, 0, 0, 0, temp.Status_ID

FROM (

SELECT mc.Lead_ID, mc.Card_ID, mc.Sold_Date, p.Lead_Name, p.Birthday,

p.Mobile_No, p.Anniversary, p.SpouseDOB, DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(SpouseDOB)), SpouseDOB) AS

TempSpouseDOB, mc.Status_ID

FROM Profiles p, dbo.func_UnionMemberCardsANDCardHistory() mc

WHERE SpouseDOB IS NOT NULL AND

mc.Lead_ID = p.Lead_ID AND

(

(@isActive = 1 AND mc.[Active] = 1) OR

(@isInactive = 1 AND mc.[Active] = 0)

) AND

mc.CardType_ID IN (SELECT *

FROM @cardTypeTblTemp) AND

(

(

@isIssued = 1 AND

mc.Status_ID = 5 ) OR

(@isCancel = 1 AND (mc.Status_ID = 6 and mc.Type = 3)) OR

(

(

@isExp = 0 AND

( mc.Status_ID <> 7 or

mc.[Expiry_Date] >= GETDATE())

) OR

(

@isExp = 1 AND

(mc.Status_ID = 7 OR

(mc.[Expiry_Date] < GETDATE()))

)

) OR

(@isRef = 1 AND (mc.Type = 2 and mc.Status_ID = 8)) OR

(@isReprint = 1 AND mc.[Type] = 1)

)

) temp

WHERE TempSpouseDOB BETWEEN DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(@before)), @before) AND DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(@after)), @after)

END

ELSE

BEGIN

INSERT INTO @tempTable

SELECT Lead_ID, Card_ID, Sold_Date, Lead_Name, Mobile_No, Birthday,

Anniversary, SpouseDOB, 0, 0, 1, 0, 0, 0, 0, 0, temp.Status_ID

FROM (

SELECT mc.Lead_ID, mc.Card_ID, mc.Sold_Date, p.Lead_Name, p.Birthday,

p.Mobile_No, p.Anniversary, p.SpouseDOB, DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(SpouseDOB)), SpouseDOB) AS

TempSpouseDOB, mc.Status_ID

FROM Profiles p,MemberCards mc

WHERE SpouseDOB IS NOT NULL AND

mc.Lead_ID = p.Lead_ID AND

(

(@isActive = 1 AND mc.[Active] = 1) OR

(@isInactive = 1 AND mc.[Active] = 0)

) AND

mc.CardType_ID IN (SELECT *

FROM @cardTypeTblTemp) AND

(

(

@isIssued = 1 AND

mc.Status_ID = 5 ) OR

(@isCancel = 1 AND (mc.Status_ID = 6)) OR

(

(

@isExp = 0 AND

( mc.Status_ID <> 7 or

mc.[Expiry_Date] >= GETDATE())

) OR

(

@isExp = 1 AND

(mc.Status_ID = 7 OR

(mc.[Expiry_Date] < GETDATE()))

)

) OR

(@isRef = 1 AND (mc.Status_ID = 8)) --OR

--(@isReprint = 1 AND mc.[Type] = 1)

)

) temp

WHERE TempSpouseDOB BETWEEN DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(@before)), @before) AND DATEADD(YEAR, (YEAR(GETDATE()) - YEAR(@after)), @after)

END

END

RETURN

END
